# Web Calling

Easily integrate **Indigenius Web SDK** or use the **Indigenius Dashboard** to enable browser-based calling.

---

---

## Step 1 — Environment Setup

Go to **Settings → Environment Variables** in your app and add your Public API key:

```bash
NEXT_PUBLIC_INDIGENIUS_API_KEY=your-public-api-key
````

---

## Step 2 — Installation

<CodeGroup>
```bash npm
npm install indigenius-ai-widget
```

```bash yarn
yarn add indigenius-ai-widget
```

</CodeGroup>

Or via CDN:

```html
<script src="https://cdn.jsdelivr.net/npm/indigenius-ai-widget@1.0.0/dist/sdk.umd.js"></script>
```

Or official hosted script:

```html
<script type="module" src="https://widget.indigenius.ai/v1/index.js"></script>
```

## Step 3 — Quick Start

\<Tabs items={\["HTML", "React", "Vue", "Angular"]}>

<TabItem value="HTML">

```html
<indigenius-convai agent-id="your-agent-id"></indigenius-convai>

<script type="module">
  import { CdialWidget } from "indigenius-ai-widget";

  const widget = new CdialWidget({
    agentId: "your-agent-id",
    mountUI: true,
  });

  widget.init();
</script>
```

</TabItem>

<TabItem value="React">

```tsx
import { useEffect } from "react";
import { CdialWidget } from "indigenius-ai-widget";

export default function App() {
  useEffect(() => {
    const widget = new CdialWidget({
      agentId: "your-agent-id",
      mountUI: true,
    });
    widget.init();

    return () => widget.destroy();
  }, []);

  return <div>Your React App</div>;
}
```

</TabItem>

<TabItem value="Vue">

```vue
<template>
  <div id="app">Your Vue App</div>
</template>

<script setup>
import { onMounted, onBeforeUnmount } from "vue";
import { CdialWidget } from "indigenius-ai-widget";

let widget;

onMounted(() => {
  widget = new CdialWidget({
    agentId: "your-agent-id",
    mountUI: true,
  });
  widget.init();
});

onBeforeUnmount(() => {
  widget.destroy();
});
</script>
```

</TabItem>

<TabItem value="Angular">

```ts
import { Component, OnInit, OnDestroy } from "@angular/core";
import { CdialWidget } from "indigenius-ai-widget";

@Component({
  selector: "app-root",
  template: "<div>Your Angular App</div>",
})
export class AppComponent implements OnInit, OnDestroy {
  widget: any;

  ngOnInit() {
    this.widget = new CdialWidget({
      agentId: "your-agent-id",
      mountUI: true,
    });
    this.widget.init();
  }

  ngOnDestroy() {
    this.widget.destroy();
  }
}
```

</TabItem>

</Tabs>

---

## Step 4 — Call Control

Start or end a call programmatically:

```js
widget.toggleCall();
```

---

## Step 5 — Event Listeners

```js
widget.on("dialing", () => console.log("Dialing..."));
widget.on("connected", () => console.log("Connected"));
widget.on("ended", () => console.log("Ended"));
widget.on("error", (err) => console.error("Error:", err));
```

---

## Step 6 — Destroy

```js
widget.destroy();
```
